21 void afficheMatriceMingl (const CMatrice & Mat, MinGL & window) {

27  window << nsShape::Circle(nsGraphics::Vec2D(j * 50 + 25, i * 50 + 25), 25, nsGraphics::KRed); // Si l'élément = 2, alors la couleur sera rouge
