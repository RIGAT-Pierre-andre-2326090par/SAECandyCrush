23 void afficheMatriceMingl (const CMatrice & Mat, MinGL & window) {
24  for (size_t i = 0 ; i < Mat.size() ; ++i) {
25  for ( size_t j = 0 ; j < Mat[i].size() ; ++j) {

27  window << nsShape::Circle(nsGraphics::Vec2D(j * 50 + 25, i * 50 + 25), 25, nsGraphics::KCyan); // Si l'élément = 1, alors la couleur sera cyan
28  else if (Mat[i][j]==2)
29  window << nsShape::Circle(nsGraphics::Vec2D(j * 50 + 25, i * 50 + 25), 25, nsGraphics::KRed); // Si l'élément = 2, alors la couleur sera rouge
30  else if (Mat[i][j]==3)
31  window << nsShape::Circle(nsGraphics::Vec2D(j * 50 + 25, i * 50 + 25), 25, nsGraphics::KGreen); // Si l'élément = 3, alors la couleur sera verte
32  else if (Mat[i][j]==4)
33  window << nsShape::Circle(nsGraphics::Vec2D(j * 50 + 25, i * 50 + 25), 25, nsGraphics::KYellow); // Si l'élément = 4, alors la couleur sera jaune

35  window << nsShape::Circle(nsGraphics::Vec2D(j * 50 + 25, i * 50 + 25), 25, nsGraphics::KMagenta); // Si l'élément = 5, alors la couleur sera magenta

37  window << nsShape::Circle(nsGraphics::Vec2D(j * 50 + 25, i * 50 + 25), 25, nsGraphics::KBlue); // Si l'élément = 6, alors la couleur sera bleu
38  }
39  }
40 }
